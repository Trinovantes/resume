	\ifdefempty{\company}{}{
		\vspace{-2pt}%
		\textcolor{dim}{\footnotesize \company \hspace*{\fill} \location} \\

	}

        \responsibilities%

\newcommand{\Internship}[3]{
	\def\dates{#1}%
	\def\title{#2}%
	\def\company{#3}%

	\LabelItem{\company}{%
		\title \hfill \textcolor{dim}{\dates}%
	}[1]
}

\newcommandx\LabelItem[3][3=0]{
	\def\Label{#1}%
	\def\Item{#2}%
	\def\Depth{#3}%

	\item \parbox{\LeftOffset - \dimexpr(\Padding * \Depth)}{\Label} \Item  \\
}
